## BLOQUE 2 — `store/ThemeProvider.tsx` (CREAR)

### Contrato del hook

```ts
type ColorMode = 'light' | 'dark' | 'system'

interface ThemeContext {
  colorMode: ColorMode          // preferencia guardada ('system' = seguir SO)
  isDark: boolean               // valor resuelto (usado para renderizar)
  theme: typeof colors          // objeto de colores activo (light o dark)
  toggleTheme: () => void       // alterna light ↔ dark (si era 'system', fija el opuesto del SO)
  setColorMode: (mode: ColorMode) => void  // fija modo explícito
}
```

### Implementación

```ts
import createContextHook from '@nkzw/create-context-hook'
import AsyncStorage from '@react-native-async-storage/async-storage'
import { useCallback, useEffect, useMemo, useState } from 'react'
import { useColorScheme } from 'react-native'
import colors from '@/constants/colors'
import { darkColors } from '@/constants/colors'

const STORAGE_KEY = 'walvy_theme_mode'

export const [ThemeProvider, useTheme] = createContextHook(() => {
  const systemScheme = useColorScheme()                    // 'light' | 'dark' | null
  const [colorMode, setColorModeState] = useState<ColorMode>('system')
  const [hydrated, setHydrated] = useState(false)

  // Cargar preferencia persistida
  useEffect(() => {
    AsyncStorage.getItem(STORAGE_KEY).then((stored) => {
      if (stored === 'light' || stored === 'dark' || stored === 'system') {
        setColorModeState(stored)
      }
      setHydrated(true)
    })
  }, [])

  const isDark = useMemo(() => {
    if (colorMode === 'system') return systemScheme === 'dark'
    return colorMode === 'dark'
  }, [colorMode, systemScheme])

  const theme = isDark ? darkColors : colors

  const setColorMode = useCallback((mode: ColorMode) => {
    setColorModeState(mode)
    AsyncStorage.setItem(STORAGE_KEY, mode)
  }, [])

  const toggleTheme = useCallback(() => {
    // Si está en 'system', fija el modo opuesto al actual del SO
    const next = isDark ? 'light' : 'dark'
    setColorMode(next)
  }, [isDark, setColorMode])

  return useMemo(
    () => ({ colorMode, isDark, theme, toggleTheme, setColorMode, hydrated }),
    [colorMode, isDark, theme, toggleTheme, setColorMode, hydrated]
  )
})
```

### Reglas de implementación
- `hydrated` evita flash antes de leer AsyncStorage; `_layout.tsx` puede mostrar `null` o el splash hasta que sea `true`.
- `useColorScheme` de `react-native` se llama dentro del hook — no duplicar en consumidores.
- No usar `useReducer` ni Redux; el `useState` par `colorMode` + `useMemo` de resolución es suficiente.

```tsx
import { useTheme } from '@/store/ThemeProvider'
import { Switch } from 'react-native'

// dentro del componente:
const { isDark, toggleTheme, theme } = useTheme()

// fila del toggle:
<View style={styles.row}>
  <Text style={[styles.rowLabel, { color: theme.textPrimary }]}>
    Modo oscuro
  </Text>
  <Switch
    value={isDark}
    onValueChange={toggleTheme}
    trackColor={{ false: theme.border, true: theme.oceanTeal }}
    thumbColor={theme.white}
  />
</View>
```

**Reglas de diseño:**
- `trackColor.true` → `theme.oceanTeal`
- `trackColor.false` → `theme.border`
- `thumbColor` → `theme.white`
- Ningún hex literal. Importar `{ spacing }` desde `@/constants/theme` para márgenes.
- El toggle va en la sección "Cuenta" (junto a los items de navegación), no en "Sesión".
